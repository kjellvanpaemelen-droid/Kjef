Integratie Arduino Uno op kot:

Op kot in de living is het bij ons donker en staat de chauffage vaak onnodig te verwarmen. We zouden dus de Dramco Uno kunnen integreren met de thermostaat. Wanneer de temperatuur onder de 18 graden gaat, moet hij bijverwarmen indien het licht uit is (en er dus niemand op kot is). Wanneer het licht aan is en we dus op kot zijn, mag de temperatuur in de kamer naar 20,5°C gebracht worden. De code zou er als volgt uit kunnen zien: 


            const int lichtPin = 2;
            const int relaisPin = 8;
            const int tempPin = A0;
            
            float temp = 0.0;
            float setpoint = 18.0;
            
            void setup() {
              pinMode(lichtPin, INPUT);
              pinMode(relaisPin, OUTPUT);
              Serial.begin(9600);
            }
            
            float leesTemperatuur() {
              int raw = analogRead(tempPin);
              float voltage = raw * (5.0 / 1023.0);
              float tempC = (voltage - 0.5) * 100.0;
              return tempC;
            }
            
            void loop() {
              bool lichtAan = digitalRead(lichtPin) == HIGH;
              temp = leesTemperatuur();
            
              if (lichtAan) {
                setpoint = 20.5;
              } else {
                setpoint = 18.0;
              }
            
              if (temp < setpoint - 0.2) {
                digitalWrite(relaisPin, HIGH);
              } else {
                if (temp > setpoint + 0.2) {
                  digitalWrite(relaisPin, LOW);
                }
              }
            
              Serial.print("Temp: ");
              Serial.print(temp);
              Serial.print(" °C | Licht: ");
              Serial.print(lichtAan ? "Aan" : "Uit");
              Serial.print(" | Setpoint: ");
              Serial.println(setpoint);
            
              delay(500);
            }
